Required Airtable fields

Create these fields in your Airtable Time Log table: Date, Week Starting, Month, Day, Start Time, End Time, Project, Task, Minutes, Hours, Notes, Created At, Entry ID. Use text fields for Start Time and End Time, number fields for Minutes and Hours, and date/date-time fields for Date, Week Starting and Created At.

Airtable proxy script

Deploy this Google Apps Script as a web app, then paste the web app URL into the Airtable proxy field above. Keep your Airtable token inside Apps Script only, not inside this HTML file.

const AIRTABLE_TOKEN = "paste_your_airtable_personal_access_token_here";
const AIRTABLE_BASE_ID = "paste_your_airtable_base_id_here";
const AIRTABLE_TABLE_NAME = "Time Log";

function doPost(e) {
  const data = JSON.parse(e.postData.contents);

  const payload = {
    fields: {
      "Date": data.date || "",
      "Week Starting": data.weekStarting || "",
      "Month": data.month || "",
      "Day": data.day || "",
      "Start Time": data.startTime || "",
      "End Time": data.endTime || "",
      "Project": data.project || "",
      "Task": data.task || "",
      "Minutes": Number(data.minutes || 0),
      "Hours": Number(data.hours || 0),
      "Notes": data.notes || "",
      "Created At": data.createdAt || new Date().toISOString(),
      "Entry ID": data.id || ""
    }
  };

  const response = UrlFetchApp.fetch(
    `https://api.airtable.com/v0/${AIRTABLE_BASE_ID}/${encodeURIComponent(AIRTABLE_TABLE_NAME)}`,
    {
      method: "post",
      headers: {
        Authorization: `Bearer ${AIRTABLE_TOKEN}`,
        "Content-Type": "application/json"
      },
      payload: JSON.stringify(payload),
      muteHttpExceptions: true
    }
  );

  const statusCode = response.getResponseCode();
  const responseText = response.getContentText();

  return ContentService
    .createTextOutput(JSON.stringify({
      status: statusCode >= 200 && statusCode < 300 ? "success" : "error",
      statusCode,
      airtableResponse: responseText
    }))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet() {
  return ContentService
    .createTextOutput("TexTrack Airtable endpoint is running.")
    .setMimeType(ContentService.MimeType.TEXT);
}
